# 数据域封装

数据域私有可以保护数据，并且使类易于维护。

```cpp
#include <iostream>

using namespace std;

class Point
{
private:
    double x;
    double y;

public:
    Point()
    {
        x = 0;
        y = 0;
    }

    Point(double x, double y)
    {
        this->x = x;
        this->y = y;
    }

    double getX()
    {
        return x;
    }

    double getY()
    {
        return y;
    }

    void setX(double x)
    {
        this->x = x;
    }

    void setY(double y)
    {
        this->y = y;
    }
};

int main(int argc, char const *argv[])
{
    Point p1;
    cout << "x :" << p1.getX() << ", y: " << p1.getY() << "." << endl;
    Point p2(100, 200);
    cout << "x :" << p2.getX() << ", y: " << p2.getY() << "." << endl;
    p2.setX(2.33);
    p2.setY(6.66);
    cout << "x :" << p2.getX() << ", y: " << p2.getY() << "." << endl;
    return 0;
}
```
