# 静态成员

静态变量和静态函数无需创建对象即可访问。

```cpp
#include <iostream>

using namespace std;

class Point
{
private:
    double x;
    double y;
    static double z;

public:
    Point()
    {
        x = 0;
        y = 0;
    }

    Point(double x, double y)
    {
        this->x = x;
        this->y = y;
    }

    double getX()
    {
        return x;
    }

    double getY()
    {
        return y;
    }

    static double getZ()
    {
        return z;
    }

    void setX(double x)
    {
        this->x = x;
    }

    void setY(double y)
    {
        this->y = y;
    }

    static void setZ(double z)
    {
        Point::z = z;
    }
};

double Point::z = 0;

int main(int argc, char const *argv[])
{
    cout << Point::getZ() << endl;
    Point::setZ(123.456);
    cout << Point::getZ() << endl;
    return 0;
}
```
