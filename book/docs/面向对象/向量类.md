# 向量类

C++ 提供了向量类，它比数组更灵活，大小可以按需自动增长。

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char const *argv[])
{
    vector<int> arr;
    for (int i = 0; i < 10; i++)
    {
        arr.push_back(i);
    }
    for (int i = 0; i < arr.size(); i++)
    {
        cout << arr[i] << " ";
    }
    cout << endl;
    return 0;
}
```

执行下面的代码，删除向量最后一个元素。

```cpp
#include <iostream>

        arr.push_back(i);
    }

    {
        cout << arr[i] << " ";
    }
    cout << endl;
    return 0;
}
```

执行下面的代码，交换两个向量数据。

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char const *argv[])
{
    vector<int> arr1, arr2;
    for (int i = 0; i < 10; i++)
    {
        arr1.push_back(i);
    }
    for (int i = 0; i < arr1.size(); i++)
    {
        cout << arr1[i] << " ";
    }
    cout << endl;
    for (int i = 9; i >= 0; i--)
    {
        arr2.push_back(i);
    }
    for (int i = 0; i < arr2.size(); i++)
    {
        cout << arr2[i] << " ";
    }
    cout << endl;
    arr1.swap(arr2);
    for (int i = 0; i < arr1.size(); i++)
    {
        cout << arr1[i] << " ";
    }
    cout << endl;
    for (int i = 0; i < arr2.size(); i++)
    {
        cout << arr2[i] << " ";
    }
    cout << endl;
    return 0;
}
```

执行下面的代码，删除向量中所有元素。

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char const *argv[])
{
    vector<int> arr;
    for (int i = 0; i < 10; i++)
    {
        arr.push_back(i);
    }
    for (int i = 0; i < arr.size(); i++)
    {
        cout << arr[i] << " ";
    }
    cout << endl;
    arr.clear();
    cout << arr.empty() << endl;

}
```

二维数组可以用向量的向量来表示。

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char const *argv[])
{
    vector<vector<int>> matrix(3);
    for (int i = 0; i < 3; i++)
    {
        matrix[i] = vector<int>(3);
        for (int j = 0; j < 3; j++)
        {
            matrix[i][j] = i + j;
        }
    }
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
```
