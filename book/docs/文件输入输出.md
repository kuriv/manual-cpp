# 文件输入输出

执行下面的代码，打开指定文件并读取数据。

```cpp
#include <iostream>
#include <fstream>

using namespace std;

int main(int argc, char const *argv[])
{
    ifstream file;
    file.open("index.txt");
    if (file.fail())
    {
        cout << "Count not load file." << endl;
        return -1;
    }
    int x, y, z;
    file >> x >> y >> z;
    cout << x << endl;
    cout << y << endl;
    cout << z << endl;
    file.close();
    return 0;
}
```

执行下面的代码，打开指定文件并写入数据。

```cpp
#include <iostream>
#include <fstream>

using namespace std;

int main(int argc, char const *argv[])
{
    ofstream file("index.txt");
    if (file.fail())
    {
        cout << "Count not load file." << endl;
        return -1;
    }
    file << 233 << " " << 6.66 << " " << "Hello World!";
    file.close();
    return 0;
}
```

执行下面的代码，读入指定文件中所有数据。

```cpp
#include <iostream>
#include <fstream>

using namespace std;

int main(int argc, char const *argv[])
{
    ifstream file("index.txt");
    if (file.fail())
    {
        cout << "Count not load file." << endl;
        return -1;
    }
    int num;
    while (!file.eof())
    {
        file >> num;
        cout << num << " ";
    }
    cout << endl;
    return 0;
}
```

执行下面的代码，对指定的文件进行输入和输出。

```cpp
#include <iostream>
#include <fstream>

using namespace std;

int main(int argc, char const *argv[])
{
    fstream file("index.txt", ios::in);
    if (file.fail())
    {
        cout << "Count not load file." << endl;
        return -1;
    }
    string str;
    file >> str;
    cout << str << endl;
    file.close();
    file.open("index.txt", ios::out | ios::app);
    file << "Welcome!";
    file.close();
    return 0;
}
```
