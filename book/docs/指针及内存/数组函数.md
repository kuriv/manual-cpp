# 数组函数

执行下面的代码，返回指向数组中最小和最大元素的指针。

```cpp
#include <iostream>
#include <algorithm>

using namespace std;

int main(int argc, char const *argv[])
{
    int arr[] = {4, 2, 3, 6, 5, 1};
    int *min = min_element(arr, arr + 6);
    cout << *min << endl;
    int *max = max_element(arr, arr + 6);
    cout << *max << endl;
    return 0;
}
```

执行下面的代码，对数组进行随机洗牌。

```cpp
#include <iostream>
#include <algorithm>

using namespace std;

void print(int arr[], int n)
{
    for (int i = 0; i < n; i++)
    {
        cout << *(arr + i) << " ";
    }
    cout << endl;
}

int main(int argc, char const *argv[])
{
    int arr[] = {4, 2, 3, 6, 5, 1};
    print(arr, 6);
    random_shuffle(arr, arr + 6);
    print(arr, 6);
    return 0;
}
```

执行下面的代码，对数组进行排序。

```cpp
#include <iostream>
#include <algorithm>

using namespace std;

void print(int arr[], int n)
{
    for (int i = 0; i < n; i++)
    {
        cout << *(arr + i) << " ";
    }
    cout << endl;
}

int main(int argc, char const *argv[])
{
    int arr[] = {4, 2, 3, 6, 5, 1};
    print(arr, 6);
    sort(arr, arr + 6);
    print(arr, 6);
    return 0;
}
```
