# 动态对象

执行下面的代码，在堆中动态创建对象。

```cpp
#include <iostream>

using namespace std;

class Point
{
private:
    double x;
    double y;

public:
    Point()
    {
        x = 0;
        y = 0;
    }

    Point(double x, double y)
    {
        this->x = x;
        this->y = y;
    }

    double getX()
    {
        return x;
    }

    double getY()
    {
        return y;
    }

    void setX(double x)
    {
        this->x = x;
    }

    void setY(double y)
    {
        this->y = y;
    }
};

int main(int argc, char const *argv[])
{
    Point *p1 = new Point();
    delete p1;
    Point *p2 = new Point;
    delete p2;
    Point *p3 = new Point(1.23, 4.56);
    cout << "x :" << p3->getX() << ", y: " << p3->getY() << "." << endl;
    cout << "x :" << (*p3).getX() << ", y: " << (*p3).getY() << "." << endl;
    delete p3;
    return 0;
}
```
