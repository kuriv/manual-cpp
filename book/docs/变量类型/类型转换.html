类型转换

C++ 允许通过转换运算符把数据由一种数据类型显式转换为另一种数据类型。

#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    cout << static_cast<int>(6.66) << endl;
    cout << static_cast<float>(1) / 2 << endl;
    return 0;
}
