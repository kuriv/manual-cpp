布尔类型

布尔类型

保存布尔值的变量称为布尔变量。

#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    bool flag = true;
    cout << flag << endl;
    flag = false;
    cout << flag << endl;
    return 0;
}
