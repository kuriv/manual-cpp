# 运算符重载

C++ 中大部分运算符都被定义为函数，用于执行一定的操作。

```cpp
#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    string str1("Hello!");
    string str2("Welcome!");
    cout << str1.operator[](0) << endl;
    cout << str1.operator=(str2) << endl;
    cout << str1.operator+=(str2) << endl;
    cout << str1.operator=("Hello!") << endl;
    cout << operator<=(str1, str2) << endl;
    return 0;
}
```

执行下面的代码，定义运算符函数。

```cpp
#include <iostream>

using namespace std;

class Point
{
private:
    double x;
    double y;

public:
    Point()
    {
        this->x = 0;
        this->y = 0;
    }

    Point(double x, double y)
    {
        this->x = x;
        this->y = y;
    }

    double getX()
    {
        return this->x;
    }

    double getY()
    {
        return this->y;
    }

    void setX(double x)
    {
        this->x = x;
    }

    void setY(double y)
    {
        this->y = y;
    }

    Point operator+(const Point p)
    {
        return Point(this->x + p.x, this->y + p.y);
    }
};

int main(int argc, char const *argv[])
{
    Point p1(100, 200);
    cout << "x :" << p1.getX() << ", y: " << p1.getY() << "." << endl;
    Point p2(1.23, 4.56);
    cout << "x :" << p2.getX() << ", y: " << p2.getY() << "." << endl;
    Point p3 = p1.operator+(p2);
    cout << "x :" << p3.getX() << ", y: " << p3.getY() << "." << endl;
    Point p4 = p1 + p2;
    cout << "x :" << p4.getX() << ", y: " << p4.getY() << "." << endl;
    return 0;
}
```
