# 对象数组

C++ 可以创建基本类型或字符串数组，同样可以创建对象的数组。

```cpp
#include <iostream>

using namespace std;

class Point
{
private:
    double x;
    double y;

public:
    Point()
    {
        this->x = 0;
        this->y = 0;
    }

    Point(double x, double y)
    {
        this->x = x;
        this->y = y;
    }

    double getX()
    {
        return this->x;
    }

    double getY()
    {
        return this->y;
    }

    void setX(double x)
    {
        this->x = x;
    }

    void setY(double y)
    {
        this->y = y;
    }
};

int main(int argc, char const *argv[])
{
    Point arr1[3];
    cout << "x :" << arr1[0].getX() << ", y: " << arr1[0].getY() << "." << endl;
    cout << "x :" << arr1[1].getX() << ", y: " << arr1[1].getY() << "." << endl;
    cout << "x :" << arr1[2].getX() << ", y: " << arr1[2].getY() << "." << endl;
    Point arr2[3] = {Point(100, 200), Point(233, 6.66), Point(1.23, 4.56)};
    cout << "x :" << arr2[0].getX() << ", y: " << arr2[0].getY() << "." << endl;
    cout << "x :" << arr2[1].getX() << ", y: " << arr2[1].getY() << "." << endl;
    cout << "x :" << arr2[2].getX() << ", y: " << arr2[2].getY() << "." << endl;
    return 0;
}
```
