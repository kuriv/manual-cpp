# 继承

C++ 继承允许声明一个基类，并随后将其扩展为派生类。

```cpp
#include <iostream>

using namespace std;

class Animal
{
public:
    void call()
    {
        cout << "Just kidding." << endl;
    }
};

class Cat : public Animal
{
public:
    void meow()
    {
        cout << "Meow." << endl;
    }
};

int main(int argc, char const *argv[])
{
    Cat c;
    c.call();
    c.meow();
    return 0;
}
```

在基类中定义的函数能够在派生类中被重新定义。

```cpp
#include <iostream>

using namespace std;

class Animal
{
public:
    void call()
    {
        cout << "Just kidding." << endl;
    }
};

class Cat : public Animal
{
public:
    void call()
    {
        cout << "Meow." << endl;
    }
};

int main(int argc, char const *argv[])
{
    Cat c;
    c.call();
    return 0;
}
```

公有成员可以被任意其他类所访问，私有成员只能在类内或通过友元函数和友元类访问，而保护成员只能在类内或派生类访问。

```cpp
#include <iostream>

using namespace std;

class Animal
{
protected:
    string name;

public:
    Animal()
    {
        this->name = "Anonymous";
    }
};

class Cat : public Animal
{
public:
    void call()
    {
        cout << this->name << endl;
    }
};

int main(int argc, char const *argv[])
{
    Cat c;
    c.call();
    return 0;
}
```
