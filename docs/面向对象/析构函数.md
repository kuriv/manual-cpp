# 析构函数

当一个对象销毁时将自动调用该析构函数。

```cpp
#include <iostream>

using namespace std;

class Point
{
private:
    double x;
    double y;

public:
    Point()
    {
        this->x = 0;
        this->y = 0;
    }

    Point(double x, double y)
    {
        this->x = x;
        this->y = y;
    }

    ~Point()
    {
        cout << "destroyed." << endl;
    }

    double getX()
    {
        return this->x;
    }

    double getY()
    {
        return this->y;
    }

    void setX(double x)
    {
        this->x = x;
    }

    void setY(double y)
    {
        this->y = y;
    }
};

int main(int argc, char const *argv[])
{
    Point p1;
    Point p2(1.23, 4.56);
    return 0;
}
```
