# 友元函数

可以通过定义友元类或友元函数，使其能够访问其他类中的私有成员。

```cpp
#include <iostream>

using namespace std;

class Point
{
private:
    double x;
    double y;

public:
    Point()
    {
        this->x = 0;
        this->y = 0;
    }

    Point(double x, double y)
    {
        this->x = x;
        this->y = y;
    }

    double getX()
    {
        return this->x;
    }

    double getY()
    {
        return this->y;
    }

    void setX(double x)
    {
        this->x = x;
    }

    void setY(double y)
    {
        this->y = y;
    }

    friend class Rect;
};

class Rect
{
public:
    static void print()
    {
        Point p;
        p.x = 100;
        p.y = 200;
        cout << "x :" << p.x << ", y: " << p.y << "." << endl;
    }
};

int main(int argc, char const *argv[])
{
    Rect::print();
    return 0;
}
```
