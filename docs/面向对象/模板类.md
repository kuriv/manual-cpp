# 模板类

执行下面的代码，定义具有通用类型的类。

```cpp
#include <iostream>

using namespace std;

template<typename T>
class Point
{
private:
    T x;
    T y;

public:
    Point()
    {
        this->x = 0;
        this->y = 0;
    }

    Point(T x, T y)
    {
        this->x = x;
        this->y = y;
    }

    T getX()
    {
        return this->x;
    }

    T getY()
    {
        return this->y;
    }

    void setX(T x)
    {
        this->x = x;
    }

    void setY(T y)
    {
        this->y = y;
    }
};

int main(int argc, char const *argv[])
{
    Point<int> p1;
    cout << "x :" << p1.getX() << ", y: " << p1.getY() << "." << endl;
    Point<int> p2(100, 200);
    cout << "x :" << p2.getX() << ", y: " << p2.getY() << "." << endl;
    Point<double> p3(1.23, 4.56);
    cout << "x :" << p3.getX() << ", y: " << p3.getY() << "." << endl;
    p3.setX(233);
    p3.setY(666);
    cout << "x :" << p3.getX() << ", y: " << p3.getY() << "." << endl;
    return 0;
}
```
